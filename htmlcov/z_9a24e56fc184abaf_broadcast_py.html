1"""Live-broadcast identifiers & URLs (docs/broadcast-urls.md) — the light, Qt-free helpers shared by 

2the Timer, the Reports settings, and the publish path (cozer/app/live.py). 

3 

4Kept deliberately import-cheap (only ``re`` + ``datetime``): the GUI thread builds viewer links and 

5validates slugs through here without pulling in the report stack (weasyprint), which ``live.py`` does. 

6 

7The *storage split* (docs/broadcast-urls.md §3): the **event name + channel live in the event** (they 

8travel with the .coz, so a different event → a different URL → no multi-instance clash), while the 

9server URL + publish secret live in cozer config (the secret must NEVER reach the .coz — its content 

10is embedded verbatim in bug reports). 

11""" 

12import re 

13from datetime import datetime 

14 

15DEFAULT_CHANNEL = "a" 

16_SLUG_RE = re.compile(r"^[a-z0-9][a-z0-9-]*$") 

17 

18 

19def slugify(text): 

20 """Coerce free text to a broadcast slug: lowercase; spaces/underscores/dots → hyphens; drop 

21 anything else; collapse and trim hyphens. Returns ``""`` if nothing usable survives (the caller 

22 supplies a default). E.g. ``"Harku 2026"`` → ``"harku-2026"``, ``"A"`` → ``"a"``, ``"_x"`` → ``"x"``.""" 

23 s = re.sub(r"[\s_.]+", "-", (text or "").strip().lower()) 

24 s = re.sub(r"[^a-z0-9-]+", "", s) 

25 return re.sub(r"-{2,}", "-", s).strip("-") 

26 

27 

28def is_slug(text): 

29 """True if ``text`` is already a valid slug (``^[a-z0-9][a-z0-9-]*$``) — matches the server's route 

30 regex, so what cozer accepts is exactly what the server will route.""" 

31 return bool(_SLUG_RE.match(text or "")) 

32 

33 

34def default_event_name(now=None): 

35 """The default broadcast name when the organizers haven't picked one: the current month+year 

36 digits, e.g. July 2026 → ``"0726"`` (docs/broadcast-urls.md §1). ``now`` is injectable for tests.""" 

37 return (now or datetime.now()).strftime("%m%y") 

38 

39 

40def event_name(eventdata): 

41 """The broadcast event name stored in the event, or the ``MMYY`` default (never blank).""" 

42 b = (eventdata or {}).get("broadcast") or {} 

43 return b.get("eventname") or default_event_name() 

44 

45 

46def event_channel(eventdata): 

47 """The broadcast channel stored in the event, or the ``"a"`` default (never blank).""" 

48 b = (eventdata or {}).get("broadcast") or {} 

49 return b.get("channel") or DEFAULT_CHANNEL 

50 

51 

52def set_broadcast(eventdata, eventname, channel): 

53 """Store the (slugified) event name + channel in the event. Blank input → the default. Returns the 

54 ``(eventname, channel)`` actually stored so the caller can echo the normalized values back.""" 

55 en = slugify(eventname) or default_event_name() 

56 ch = slugify(channel) or DEFAULT_CHANNEL 

57 eventdata.setdefault("broadcast", {}) 

58 eventdata["broadcast"]["eventname"] = en 

59 eventdata["broadcast"]["channel"] = ch 

60 return en, ch 

61 

62 

63def feed_path(eventname, channel): 

64 """A feed's path segment: ``<eventname>/feed/<channel>`` (docs/broadcast-urls.md §2).""" 

65 return "%s/feed/%s" % (eventname, channel) 

66 

67 

68def viewer_url(base_url, eventname, channel): 

69 """The public viewer URL for a feed: ``<base_url>/<eventname>/feed/<channel>/``.""" 

70 return "%s/%s/" % ((base_url or "").rstrip("/"), feed_path(eventname, channel))
