10def heat_label(heat_id): 

11 """A heat id as displayed in a report header. UIM 209 restart notation: ``1``→``1``, 

12 ``1r``→``1R`` (first restart), ``1R``→``1R2`` (second restart). Time-trial (``2t``) and 

13 qualification (``3q``) ids show a **bare number** (``2``/``3``) — the phase kind is already 

14 shown separately (report subtitle / column), so the ``t``/``q`` would be a redundant leak. 

15 Presentation-only; the ``R`` suffix already means "second restart", so no finals/last-heat 

16 context is needed. An unrecognized id raises (surfaced, not silently mangled).""" 

17 number, suffix = _parse_heat_id(heat_id) 

18 return "%d%s" % (number, {"r": "R", "R": "R2", "t": "", "q": ""}.get(suffix, suffix))

30def get_fullname(first, last): 

31 """Combine first/last, supporting ';'-separated multi-driver boats 

32 (faithful port of legacy reports.get_fullname).""" 

33 first, last = str(first), str(last) 

34 if ";" in first and first.count(";") > last.count(";"): 

35 last += ";" * (first.count(";") - last.count(";")) 

36 if ";" in last and first.count(";") < last.count(";"): 

37 first += ";" * (last.count(";") - first.count(";")) 

38 if ";" in first and first.count(";") == last.count(";"): 

39 return "; ".join("%s %s" % (f, l) for f, l in zip(first.split(";"), last.split(";"))) 

40 return "%s %s" % (first, last)

69def _has_distinct(values): 

70 """True if ``values`` are not all the same — the condition for showing an optional report 

71 column. Empty counts as its own value, so an all-empty column and a uniform column (every row 

72 identical, e.g. a national event's all-``EST``) are both hidden — they distinguish nothing — 

73 but a column where some rows are filled and others blank IS shown (the filled rows carry 

74 information worth printing). (owner's rule for From/club + Nationality.)""" 

75 return len({(v or "").strip() for v in values}) > 1

132def _posting_block(labels, meta): 

133 """The §209 signature block for a results sheet: ruled lines for each *assigned* signer — the 

134 OOD/Race Director (or delegate, §209/p37) and the UIM Sports Commissioner (co-signs before 

135 posting, p30 item 14). A signer with no name is omitted; if neither is set the block is empty. 

136 Inline-styled so it never touches the shared TABLE_CSS (byte-identical for the frozen legacy).""" 

137 cells = [] 

138 for role, name in ((labels["OfficeroftheDay"], meta.get("officer", "")), 

139 (labels["UIMCommissioner"], meta.get("uim_commissioner", ""))): 

140 if not (name or "").strip(): 

141 continue # only show a signer that has a name 

142 cells.append('<td style="width:44%%;border:0;padding:5.5em 0 0 0;text-align:center;' 

143 'vertical-align:bottom">' 

144 '<div style="border-top:1px solid #000;padding-top:2px;font-size:9pt">%s</div>' 

145 '<div style="font-size:8pt;color:#444">%s</div></td>' 

146 % (display(name), esc(role))) 

147 if not cells: 

148 return "" 

149 filler = ('<td style="width:12%;border:0"></td>' if len(cells) == 2 

150 else '<td style="width:56%;border:0"></td>') 

151 inner = cells[0] + filler + (cells[1] if len(cells) == 2 else "") 

152 return ('<table style="width:100%%;border:0;border-collapse:collapse;break-inside:avoid">' 

153 '<tr>%s</tr></table>' % inner)

156def document_html(orientation, labels, meta, heading, body_parts, subtitle="", posting=False): 

157 """Wrap report body (a list of HTML fragments) in a full styled document. ``subtitle`` 

158 (optional) is shown under the heading -- used for the phase-kind line (§10-E). ``posting`` adds 

159 the §209 posting metadata (a render-time "Printed on" footer stamp + the top-right "Posted at" 

160 line + the signature block); 

161 it is passed only by the native results builders, so the frozen legacy reports (posting off) are 

162 byte-identical.""" 

163 now = datetime.now() 

164 if posting: 

165 # The OOD is a signer in the posting block, so it is redundant in the footer — put the 

166 # informational render-time "Printed on" stamp in its place (footer-left); keep the 

167 # Secretary only when named. (posting reports only, so the legacy footer is untouched.) 

168 footer_left = "%s %s" % (labels["PrintedOn"], now.strftime("%Y-%m-%d %H:%M")) 

169 footer_right = (("%s /%s/" % (meta["secretary"], labels["SecretaryoftheRace"])) 

170 if (meta.get("secretary") or "").strip() else "") 

171 else: 

172 footer_left = "%s /%s/" % (meta["officer"], labels["OfficeroftheDay"]) 

173 footer_right = "%s /%s/" % (meta["secretary"], labels["SecretaryoftheRace"]) 

174 css = page_css(orientation, footer_left=footer_left, footer_center=labels["Page"], 

175 footer_right=footer_right) 

176 parts = ["<style>%s\n%s</style>" % (css, TABLE_CSS)] 

177 if posting: # top-right: "Posted at ____:____" — the time is hand-written at posting (owner) 

178 parts.append(_posted_at(labels)) 

179 parts.append('<h1 class="event-title">%s</h1>' % display(meta["title"])) 

180 parts.append('<div class="event-meta">%s &nbsp;&middot;&nbsp; %s</div>' 

181 % (display(meta["venue"]), display(meta["date"]))) 

182 parts.append('<h2 class="report-heading">%s</h2>' % display(heading)) 

183 if subtitle: 

184 parts.append('<div class="report-subtitle">%s</div>' % display(subtitle)) 

185 parts.extend(body_parts) 

186 if posting: # signature block at the foot (only for signers that have a name) 

187 block = _posting_block(labels, meta) 

188 if block: 

189 parts.append(block) 

190 return "\n".join(parts)
