94def _system_child_env(): 

95 """Environment for launching a *system* helper program (the PDF viewer, via 

96 xdg-open/open). 

97 

98 cozer's launcher (``__main__._ensure_fontconfig``) exports ``FONTCONFIG_FILE`` 

99 pointing at a private, env-specific fontconfig config to dodge a conda 

100 libfontconfig cache segfault; a conda launcher may likewise put 

101 ``$CONDA_PREFIX/lib`` on ``LD_LIBRARY_PATH``. Child processes inherit both, so 

102 a *system* viewer -- which links the system libfontconfig and system GLib -- 

103 would parse cozer's env config (spurious ``invalid attribute 'xsi:nil'`` 

104 warnings) or load incompatible conda libraries (crash). Hand system helpers a 

105 clean environment: drop cozer's fontconfig override and any conda entries from 

106 the loader path so they use their own system libraries. 

107 """ 

108 env = dict(os.environ) 

109 for var in ("FONTCONFIG_FILE", "FONTCONFIG_PATH"): 

110 env.pop(var, None) 

111 prefixes = tuple(p for p in (sys.prefix, os.environ.get("CONDA_PREFIX")) if p) 

112 for var in ("LD_LIBRARY_PATH", "LD_PRELOAD"): 

113 val = env.get(var) 

114 if not val: 

115 continue 

116 kept = [e for e in val.split(os.pathsep) 

117 if e and not any(e == p or e.startswith(p + os.sep) for p in prefixes)] 

118 if kept: 118 ↛ 121line 118 didn't jump to line 121 because the condition on line 118 was always true

119 env[var] = os.pathsep.join(kept) 

120 else: 

121 env.pop(var, None) 

122 return env
