1"""Pre-Race Technical Inspection Form (issue #30). 

2 

3Two variants of the UIM pre-race scrutineering check list, from the 2026 UIM Circuit Rulebook's own 

4recommended list (§503.02) + the reinforced-cockpit block (§509): 

5 

6 * **Cockpit** classes (reinforced §509 safety capsule — F2/F4/F 500): general items + the full §509 

7 cockpit set. 

8 * **Non-cockpit** (open) classes (F 125/F 250/OSY 400/GT15/GT30): general items + the open-boat items. 

9 

10Each checklist line is **mandatory unless marked**: a trailing italic note flags a *conditional* item 

11("if fitted", "titled events", "F4/S3 only", …) or a *recommended* one. Mandatory technical conditions 

12that a specialist already certified once (the cockpit was registered, the hull got its HIN, …) are NOT 

13physically re-inspected at the ramp — they sit in a separate **"Documents & certificates"** block where 

14the scrutineer confirms the paperwork exists. 

15 

16The operator picks the variant (two entries in the Reports combo). Class(es) selected → a multi-page PDF, 

17one page per registered boat, with Class / Boat number / Driver pre-filled; nothing selected → one blank 

18page. English-only (UIM official-form language). Article numbers are universal. 

19"""

122def _boatkey(rec): 

123 pid = rec[4] 

124 try: 

125 return (0, int(pid)) 

126 except (TypeError, ValueError): 

127 return (1, str(pid))

130def _boats_for(eventdata, classes): 

131 """(class, boat_number, driver) per registered boat in the selected classes, boat-number order. 

132 Empty when ``classes`` is falsy (nothing selected) -> caller emits one blank template page.""" 

133 if not classes: 

134 return [] 

135 by = participants_by_class(eventdata) 

136 boats = [] 

137 for cl in classes: 

138 for _key, first, last, _club, pid in sorted(by.get(cl, []), key=_boatkey): 

139 driver = get_fullname(first, last).split(";")[0].strip() 

140 boats.append((cl, str(pid), driver)) 

141 return boats

144def _build(eventdata, classes, cockpit): 

145 sections = [("General", _GENERAL)] 

146 sections.append(("Reinforced cockpit (§509)", _COCKPIT) if cockpit else ("Open boats", _OPEN)) 

147 docs = list(_DOCS_COMMON) + (list(_DOCS_COCKPIT) if cockpit else []) 

148 heading = "Pre-Race Technical Inspection — %s" % ( 

149 "Cockpit Classes" if cockpit else "Non-cockpit Classes") 

150 return {"meta": meta_of(eventdata), "labels": get_labels(eventdata), "orientation": "portrait", 

151 "cockpit": cockpit, "sections": sections, "docs": docs, "heading": heading, 

152 "fit_note": _COCKPIT_FIT if cockpit else "", "boats": _boats_for(eventdata, classes)}

155def _fields_html(cl, boat, driver): 

156 def val(x): 

157 return esc(x) if x else '<span class="ins-fbox"></span>' 

158 row = ('<div class="ins-field"><b>Class:</b> %s&nbsp;&nbsp;&nbsp;<b>Boat number:</b> %s' 

159 % (val(cl), val(boat))) 

160 if driver: 

161 row += '&nbsp;&nbsp;&nbsp;<b>Driver:</b> %s' % display(driver) 

162 return row + "</div>"

165def _item_html(name, art, cond): 

166 cond_html = ' <span class="cond">— %s</span>' % esc(cond) if cond else "" 

167 art_html = '<span class="art">%s</span>' % esc(art) if art else "" 

168 return ('<div class="ci"><span class="box"></span>' 

169 '<span class="txt">%s%s</span>%s</div>' % (esc(name), cond_html, art_html))

172def _section_html(title, items): 

173 body = "".join(_item_html(*it) for it in items) 

174 return '<div class="ins-grp">%s</div><div class="ins-cols">%s</div>' % (esc(title), body)

177def _docs_html(docs): 

178 items = "".join(_item_html(name, ref, cond) for name, ref, cond in docs) 

179 return ('<div class="ins-docs"><div class="dh">Documents &amp; certificates — confirm each exists ' 

180 '(no physical re-inspection):</div><div class="ins-cols">%s</div></div>' % items)

183def _html(model): 

184 header = ('<h1 class="event-title">%s</h1>' 

185 '<div class="event-meta">%s &nbsp;&middot;&nbsp; %s</div>' 

186 '<h2 class="report-heading">%s</h2>' 

187 % (display(model["meta"]["title"]), display(model["meta"]["venue"]), 

188 display(model["meta"]["date"]), esc(model["heading"]))) 

189 legend = ('<div class="ins-legend">Every item is mandatory unless marked in italics ' 

190 '(condition, or “recommended”). Tick each check.</div>') 

191 sections = "".join(_section_html(t, items) for t, items in model["sections"]) 

192 fit = '<div class="ins-note">%s</div>' % esc(model["fit_note"]) if model["fit_note"] else "" 

193 docs = _docs_html(model["docs"]) 

194 sig = ('<div class="ins-sig"><div class="ins-sigline"></div>' 

195 '<div class="role">Technical Officer</div></div>') 

196 

197 pages = [] 

198 for cl, boat, driver in (model["boats"] or [("", "", "")]): # [] -> one blank template page 

199 pages.append('<section class="ins-page">%s%s%s%s%s%s%s</section>' 

200 % (header, _fields_html(cl, boat, driver), legend, sections, fit, docs, sig)) 

201 

202 L, meta = model["labels"], model["meta"] 

203 css = page_css(model["orientation"], 

204 footer_left="%s /%s/" % (meta["officer"], L["OfficeroftheDay"]), 

205 footer_center=L["Page"], 

206 footer_right="%s /%s/" % (meta["secretary"], L["SecretaryoftheRace"])) 

207 return "<style>%s\n%s\n%s</style>%s" % (css, TABLE_CSS, _CSS_BODY, "".join(pages))
