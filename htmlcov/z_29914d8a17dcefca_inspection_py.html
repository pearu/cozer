1"""Pre-Race Technical Inspection Form (issue #30). 

2 

3Two variants of the UIM pre-race scrutineering check list, mirroring the 2026 UIM Circuit Rulebook's 

4own recommended list (§503.02) plus the reinforced-cockpit block (§509): 

5 

6 * **Cockpit** classes (reinforced §509 safety capsule, e.g. F2/F4/F 500): general items + the §509 

7 cockpit block. 

8 * **Non-cockpit** (open) classes (e.g. F 125/F 250/OSY 400/GT15/GT30): general items + the open-boat 

9 items (paddle, lanyard kill-cord, sponson-fin system). 

10 

11The operator chooses the variant (two entries in the Reports combo). Each form is a compact **single 

12page**. Behaviour by class selection in the Reports tab: 

13 

14 * **class(es) selected** -> a multi-page PDF with **one page per registered boat**, the *Class* and 

15 *Boat number* (and driver) pre-filled from the registration data — every boat needs a form. 

16 * **nothing selected** (``classes`` is ``None``) -> a single blank page whose *Class* and *Boat 

17 number* are filled in by hand. 

18 

19English-only (UIM official-form language); the article numbers are universal. 

20"""

111def _boatkey(rec): 

112 pid = rec[4] 

113 try: 

114 return (0, int(pid)) 

115 except (TypeError, ValueError): 

116 return (1, str(pid))

119def _boats_for(eventdata, classes): 

120 """One (class, boat_number, driver) tuple per registered boat in the selected classes, boat-number 

121 order. Empty list when ``classes`` is falsy (None / nothing selected) -> caller emits one blank 

122 template page.""" 

123 if not classes: 

124 return [] 

125 by = participants_by_class(eventdata) 

126 boats = [] 

127 for cl in classes: 

128 for _key, first, last, _club, pid in sorted(by.get(cl, []), key=_boatkey): 

129 driver = get_fullname(first, last).split(";")[0].strip() 

130 boats.append((cl, str(pid), driver)) 

131 return boats

134def _build(eventdata, classes, cockpit): 

135 sections = [("General", _GENERAL)] 

136 sections.append(("Reinforced cockpit (§509)", _COCKPIT) if cockpit else ("Open boats", _OPEN)) 

137 docs = _DOCS_HEAD + (_DOCS_COCKPIT if cockpit else []) + _DOCS_TAIL 

138 heading = "Pre-Race Technical Inspection — %s" % ( 

139 "Cockpit Classes" if cockpit else "Non-cockpit Classes") 

140 return {"meta": meta_of(eventdata), "labels": get_labels(eventdata), "orientation": "portrait", 

141 "cockpit": cockpit, "sections": sections, "docs": docs, "heading": heading, 

142 "boats": _boats_for(eventdata, classes)}

145def _fields_html(cl, boat, driver): 

146 def val(x): 

147 return esc(x) if x else '<span class="ins-fbox"></span>' 

148 row = ('<div class="ins-field"><b>Class:</b> %s&nbsp;&nbsp;&nbsp;<b>Boat number:</b> %s' 

149 % (val(cl), val(boat))) 

150 if driver: 

151 row += '&nbsp;&nbsp;&nbsp;<b>Driver:</b> %s' % display(driver) 

152 return row + "</div>"

155def _checklist_html(sections): 

156 colg = ('<colgroup><col style="width:60%"><col style="width:15%">' 

157 '<col style="width:25%"></colgroup>') 

158 head = '<tr><th>Item</th><th class="art">U.I.M. art.</th><th>Check</th></tr>' 

159 rows = [] 

160 for title, items in sections: 

161 rows.append('<tr class="grp"><td colspan="3">%s</td></tr>' % esc(title)) 

162 for item, art in items: 

163 rows.append('<tr><td>%s</td><td class="art">%s</td><td></td></tr>' % (esc(item), esc(art))) 

164 return ('<table class="ins-ck">%s<thead>%s</thead><tbody>%s</tbody></table>' 

165 % (colg, head, "".join(rows)))

168def _html(model): 

169 L, meta = model["labels"], model["meta"] 

170 header = ('<h1 class="event-title">%s</h1>' 

171 '<div class="event-meta">%s &nbsp;&middot;&nbsp; %s</div>' 

172 '<h2 class="report-heading">%s</h2>' 

173 % (display(meta["title"]), display(meta["venue"]), display(meta["date"]), 

174 esc(model["heading"]))) 

175 checklist = _checklist_html(model["sections"]) 

176 docs = ('<div class="ins-docs"><b>The driver must be able to show:</b><ul>%s</ul></div>' 

177 % "".join("<li>%s</li>" % esc(d) for d in model["docs"])) 

178 sig = ('<div class="ins-sig"><div class="ins-sigline"></div>' 

179 '<div class="role">Technical Officer</div></div>') 

180 

181 pages = [] 

182 for cl, boat, driver in (model["boats"] or [("", "", "")]): # [] -> one blank template page 

183 pages.append('<section class="ins-page">%s%s%s%s%s</section>' 

184 % (header, _fields_html(cl, boat, driver), checklist, docs, sig)) 

185 

186 css = page_css(model["orientation"], 

187 footer_left="%s /%s/" % (meta["officer"], L["OfficeroftheDay"]), 

188 footer_center=L["Page"], 

189 footer_right="%s /%s/" % (meta["secretary"], L["SecretaryoftheRace"])) 

190 return "<style>%s\n%s\n%s</style>%s" % (css, TABLE_CSS, _CSS_BODY, "".join(pages))
