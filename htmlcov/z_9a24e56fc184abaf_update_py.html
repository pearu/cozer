113def _key(v): 

114 """A comparable key for a version string, without `packaging`. Handles ``X.Y.Z`` plus an

116 m = re.match(r"^(\d+)(?:\.(\d+))?(?:\.(\d+))?(?:\.?(dev|a|alpha|b|beta|rc|c)(\d+)?)?", _clean(v)) 

117 if not m:

174def recommend(res): 

175 """The apply action for this install, given a :func:`check` result (docs/RELEASE.md §Phase 2): 

176 ``{"action", "url", "hint"}`` where ``action`` is one of 

177 

178 - ``"none"`` — up to date / offline; nothing to do. 

179 - ``"source"`` — running from a checkout: informational only (never mutate a working tree). 

180 - ``"pip"`` — the fast update: ``pip install -U --no-deps`` the release wheel (``url``). 

181 Used for a plain pip install **and** the Windows constructor install (which 

182 is a real Python env with pip); cozer's wheel is pure Python with **no** 

183 runtime deps, so this replaces only cozer's own code -- no big download, 

184 pyside6/weasyprint untouched. 

185 - ``"installer"`` — a Windows install when the release has no wheel: download + run the full 

186 installer (``url``). Also the manual fallback if the fast update ever isn't 

187 enough (e.g. the bundled libraries changed). 

188 - ``"link"`` — no usable asset: just open the release page (``url``). 

189 """ 

190 if not res.get("available") or not res.get("latest"): 

191 return {"action": "none", "url": None, "hint": ""} 

192 rel = res["latest"] 

193 kind = res.get("kind") 

194 if kind == "source": 

195 return {"action": "source", "url": rel.get("url"), "hint": "git pull && pip install -U ."} 

196 if kind in ("wheel", "windows-installer"): # both have pip -> fast wheel update 196 ↛ 200line 196 didn't jump to line 200 because the condition on line 196 was always true

197 wheel = _find_asset(rel, ".whl") 

198 if wheel:

201 # The wheel-only `releases/latest` carries no .exe; the installer lives in the fixed 

202 # `windows-installer` release (stable URL), so point there rather than at ``rel``. 

203 return {"action": "installer", "url": WINDOWS_INSTALLER_URL, "hint": "COZER-Setup-Windows.exe"} 

204 return {"action": "link", "url": rel.get("url"), "hint": "release page"} # nothing usable
