1"""Derived start order — PHASES.md §8 step 3 / §5. 

2 

3``start_order(eventdata, cl, heat)`` returns the ordered boat-id list for a heat's 

4**start / jetty positions**, DERIVED (never stored) from the previous heat's ranking:: 

5 

6 start_order(heat) = seed( previous_heat, ranking(previous_heat) ) 

7 ranking = the analyzer's finishing order (getresorder) of the previous 

8 heat's canonical (last non-empty) record. 

9 

10Base case (a phase's first heat, no predecessor) = the **participant order** in 

11Classes & Participants. Consumers are the **reports** (start list / jetty positions) 

12and the timer **ladder** (§5, owner clarification rev 29); the timer's button grid 

13stays boat-number and is NOT a consumer. 

14 

15This increment implements the base case + the **circuit heat N → N+1** transition 

16(seed by the previous number's canonical ranking). The cross-phase transitions — 

17time-trial → finals (best time) and qualification → finals (§5.1) — are a later 

18increment; a first heat with a preceding phase currently falls back to the base case. 

19Pure and read-only: it never mutates ``eventdata``. 

20""" 

21from cozer.analyzer import analyze, getresorder, rule_action_codes 

22from cozer.classes import getclass 

23from cozer.phases import class_phase_map, heat_number, phase_heat_ids 

24

26def start_order(eventdata, cl, heat): 

27 """Ordered boat-id list (strings) for class ``cl``'s ``heat`` start positions.""" 

28 ph = class_phase_map(eventdata).get(cl) 

29 if ph is not None and heat_number(heat) > 1: 

30 prev = _canonical_record(ph, heat_number(heat) - 1) 

31 if prev is not None: 

32 hid, rec = prev 

33 # analyze memoizes 'racetime' into info; copy info so start_order stays 

34 # read-only (the boats, rec[1], are not mutated -- §6.6 -- so sharing them). 

35 res = analyze(hid, [dict(rec[0]), rec[1]], eventdata.get("scoringsystem", []), 

36 rule_action_codes(eventdata)) 

37 return [str(pid) for pid in getresorder(res)] 

38 # first heat of the phase (or the predecessor is unraced/missing) -> base case 

39 return _participant_order(eventdata, cl)

42def _canonical_record(phase, number): 

43 """``(heat_id, [info, boats])`` for ``number``'s canonical record in ``phase`` — 

44 the **last non-empty** one (an empty restart is skipped, §5.2) — or ``None`` if 

45 the number has no non-empty record yet.""" 

46 hids = phase_heat_ids(phase) 

47 same = [(h, rec) for h, rec, num in zip(hids, phase.heats, phase.numbers) 

48 if num == number] 

49 for h, rec in reversed(same): 

50 if any(rec[1].values()): # rec = [info, boats]; some boat has marks 

51 return h, rec 

52 return None

55def _participant_order(eventdata, cl): 

56 """Base-case start order: the class's participants in Classes & Participants list 

57 order (the drag-reorderable lot-draw order, §5). ``cl`` may be suffixed; match on 

58 the base class, as the timer's ``class_ids`` does.""" 

59 base = getclass(cl) 

60 return [str(p[5]) for p in eventdata.get("participants", []) 

61 if len(p) > 5 and p[4] == base and str(p[5]) != ""]
