1"""Live ordering feed — the *unofficial* running order published during a race (LIVE.md, Phase 7). 

2 

3Builds a small snapshot (the "machine feed", LIVE.md §4) and publishes it to the self-hosted live 

4server (deploy/live-server, ``https://live.cozer.ee/``). Pure and headless-testable: the caller (the 

5Timer) supplies the leader-first boat order — e.g. ``timer.standings(rec)`` — so nothing here 

6imports Qt. 

7 

8Network errors are **not** swallowed here — the caller (Timer) guards them so a failed publish 

9never blocks timing (LIVE.md §5). Keeping this layer raise-on-error keeps it testable. 

10"""

22def snapshot(eventdata, cl, heat, order, updated, view=None, live=True, course=None, elapsed=None): 

23 """The unofficial live-order snapshot ``dict`` for class ``cl`` heat ``heat``. 

24 

25 ``order`` — leader-first, one item per boat. Each item is either a **boat id** (scalar) or a 

26 **standings dict** ``{"id"|"boat", "laps", "time", "finished", "laptimes"}`` (pass 

27 ``timer.standings(rec)``). When laps/time are present they flow into the feed so the viewer can 

28 show laps-completed + the time gap to the leader, and switch from the pre-start (nat/boat/surname) 

29 layout to the running layout once the field has started. 

30 ``updated`` — the publish timestamp (ISO-8601 string; the caller stamps it). 

31 ``view`` — the operator's display config (page size / dwell times); ``DEFAULT_VIEW`` if None. 

32 ``course`` — the per-lap lengths of this heat; with each boat's ``laptimes`` the viewer builds a 

33 piece-wise-linear distance model per boat (lap speed = lap length / lap time) to show a live 

34 catch-up-to-the-leader time. 

35 ``elapsed`` — the current **race-elapsed** seconds at publish (the timer's race clock); the viewer 

36 anchors its own clock to it so it can extrapolate each boat's distance forward between crossings. 

37 ``None`` when not timing (the viewer then falls back to the latest crossing — a static value). 

38 

39 The ``order`` carries the **full** field; paging is the viewer's job (LIVE.md §7). 

40 """ 

41 parts = participants_index(eventdata) 

42 nats = nationalities_index(eventdata) 

43 rows, started = [], False 

44 for i, item in enumerate(order): 

45 bid = item.get("id", item.get("boat")) if isinstance(item, dict) else item 

46 _, last, _ = parts.get((cl, str(bid)), ("", "", "")) 

47 row = {"pos": i + 1, "boat": str(bid), "surname": last, 

48 "nat": nats.get((cl, str(bid)), "")} 

49 if isinstance(item, dict): 

50 if item.get("laps") is not None: 50 ↛ 54line 50 didn't jump to line 54 because the condition on line 50 was always true

51 row["laps"] = item["laps"] 

52 if item["laps"] >= 1: 

53 started = True 

54 if item.get("time") is not None: 54 ↛ 56line 54 didn't jump to line 56 because the condition on line 54 was always true

55 row["time"] = item["time"] # cumulative seconds at the last crossing 

56 if item.get("finished"): 

57 row["finished"] = True # crossed the last lap-line -> viewer shows 🏁 

58 if item.get("laptimes"): 

59 row["laptimes"] = list(item["laptimes"]) # per-lap crossing times -> catch-up interval 

60 rows.append(row) 

61 return { 

62 "class": getclass(cl), 

63 "phase": race_kind(eventdata, cl), 

64 "heat": str(heat), 

65 "updated": updated, 

66 "unofficial": True, 

67 "live": live, 

68 "started": started, # any boat has completed >=1 lap 

69 "view": dict(view) if view else dict(DEFAULT_VIEW), 

70 "course": [c for c in course] if course else [], # per-lap lengths -> distance model 

71 "elapsed": elapsed, # race-elapsed at publish (viewer clock anchor) 

72 "order": rows, 

73 }

76def stopped(eventdata, cl, heat, updated, view=None): 

77 """A "broadcast off" snapshot — empty ``order`` + ``live: False`` — published when the operator 

78 unticks, so the viewer shows a "live stream disabled" state instead of stale positions.""" 

79 return snapshot(eventdata, cl, heat, [], updated, view=view, live=False)

82def _http_post(method, url, headers, data): # pragma: no cover - real network 

83 import urllib.request 

84 req = urllib.request.Request(url, data=data, headers=headers, method=method) 

85 with urllib.request.urlopen(req, timeout=15) as resp: 

86 return resp.status, resp.read()

89def publish_server(base_url, eventname, channel, secret, snap, transport=None): 

90 """Publish ``snap`` to the self-hosted live server: POST the snapshot JSON to 

91 ``<base_url>/_publish/<eventname>/feed/<channel>`` with the shared secret in the 

92 ``X-Publish-Secret`` header (docs/broadcast-urls.md). Fresh, no token in any URL, no rate limit. 

93 ``transport(method, url, headers, data)`` is injectable for tests. Raises on a non-2xx or network 

94 error (the Timer guards it).""" 

95 url = "%s/_publish/%s" % (base_url.rstrip("/"), feed_path(eventname, channel)) 

96 body = json.dumps(snap, ensure_ascii=False).encode("utf-8") 

97 headers = {"Content-Type": "application/json", "X-Publish-Secret": secret, "User-Agent": "cozer"} 

98 status, _ = (transport or _http_post)("POST", url, headers, body) 

99 if not 200 <= (status or 0) < 300: 

100 raise RuntimeError("live server publish failed: HTTP %s" % status) 

101 return feed_path(eventname, channel)
